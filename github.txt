# push to github
git add .
git commit -m "comment"
git remote add origin https://github.com/martinty/KTN.git
git push origin master

# pull from github
git init
git pull https://github.com/martinty/KTN.git 

# clone from github
git init
git clone https://github.com/martinty/KTN.git

# Note
Må pulle først hvis man ønsker å pushe de endringene man har gjort.
Ellers kan man bare clone.

# clone from branch
git init 
git clone -b <branch> https://github.com/martinty/KTN.git

# pull from branch
git init
git pull https://github.com/martinty/KTN.git/ <branch>

# push to branch
git add .
git commit -m "comment"
git checkout -b <branch>
git push -u  https://github.com/martinty/KTN.git <branch>
